class ArrayList {

    this.arr = [];

    this.pos = -1;
  }

  ListInit() {
    this.count = 0;
    this.pos = -1;
  }

  LInsert(data) {
    this.arr[this.count] = data;
    this.count++;
  }

  LFirst(data) {
    if (this.count === 0) {
      return false;
    }
    this.pos = 0;
    data = this.arr[0];
    return true;
  }

  LNext(data) {
    if (this.pos >= this.count - 1) {
      return false;
    }
    this.pos++;
    data = this.arr[this.pos];
    return true;
  }

  LRemove() {
    const rpos = this.pos;
    const num = this.count;
    const rdata = this.arr[rpos];

    for (let i = rpos; i < num - 1; i++) {
      this.arr[i] = this.arr[i + 1];
    }
    this.count--;
    this.pos--;
    return rdata;
  }

  LCount() {
    return this.count;
  }
