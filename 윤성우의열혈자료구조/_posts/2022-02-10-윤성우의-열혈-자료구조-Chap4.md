class LinkedList {
  constructor() {
    this.head = new Node();
    this.head.next = null;
    this.count = 0;
    this.before = null;
    this.cur = null;
  }

  ListInit() {
    this.head = new Node();
    this.head.next = null;
    this.count = 0;
    this.cur = null;
  }

  LInsert(data) {
    let newNode = new Node();
    newNode.data = data;

    newNode.next = this.head.next;
    this.head.next = newNode;

    this.count++;
  }

  LFirst() {
    this.before = this.head;
    this.cur = this.head.next;
    return this.cur.data;
  }

  LNext() {
    if (this.cur.next === null) {
      return false;
    }

    this.before = this.cur;
    this.cur = this.cur.next;

    return this.cur.data;
  }

  LRemove() {
    let rpos = this.cur;
    let rdata = rpos.data;

    this.before.next = this.cur.next;
    this.cur = this.before;

    this.count--;
    return rdata;
  }

  LCount() {
    return this.count;
  }
}
