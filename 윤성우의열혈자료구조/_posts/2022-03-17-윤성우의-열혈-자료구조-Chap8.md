  let op1, op2;

  if (getLeftSubTree(bt) === null && getRightSubTree(bt) === null) {
    return getData(bt);
  }
  op1 = evaluateExpTree(getLeftSubTree(bt));
  op2 = evaluateExpTree(getRightSubTree(bt));

  switch (getData(bt)) {
    case "+":
      return op1 + op2;
    case "-":
      return op1 - op2;
    case "*":
      return op1 * op2;
    case "/":
      return op1 / op2;
  }
  return 0;
}
